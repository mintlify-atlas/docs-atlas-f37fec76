---
title: Get Workflow Instance
api: GET /api/v1/workflow-instances/{workflow_instance_id}
---

Retrieves detailed information about a specific workflow instance, including all steps, actions, and current state.

## Authentication

Requires Bearer token authentication.

## Path Parameters

<ParamField path="workflow_instance_id" type="UUID" required>
  The unique identifier of the workflow instance to retrieve.
</ParamField>

## Response

<ResponseField name="item" type="object">
  The complete workflow instance details.
  
  <ResponseField name="item.id" type="UUID">
    Unique identifier for the workflow instance.
  </ResponseField>
  
  <ResponseField name="item.workflowDefinitionId" type="UUID">
    The workflow definition this instance belongs to.
  </ResponseField>
  
  <ResponseField name="item.workflowDefinitionKey" type="string">
    The key of the workflow definition.
  </ResponseField>
  
  <ResponseField name="item.workflowDefinitionName" type="string">
    Human-readable name of the workflow.
  </ResponseField>
  
  <ResponseField name="item.workflowVersionId" type="UUID">
    The specific version of the workflow definition.
  </ResponseField>
  
  <ResponseField name="item.businessKey" type="string | null">
    Optional business identifier.
  </ResponseField>
  
  <ResponseField name="item.runNumber" type="integer">
    Sequential run number for this business key.
  </ResponseField>
  
  <ResponseField name="item.status" type="string">
    Current workflow status: `running`, `waiting`, `paused`, `completed`, `rejected`, `failed`, or `cancelled`.
  </ResponseField>
  
  <ResponseField name="item.currentStepCode" type="string | null">
    Code of the current step.
  </ResponseField>
  
  <ResponseField name="item.currentStepLabel" type="string | null">
    Label of the current step.
  </ResponseField>
  
  <ResponseField name="item.startedBy" type="string">
    User ID of the workflow initiator.
  </ResponseField>
  
  <ResponseField name="item.startedAt" type="string">
    ISO 8601 timestamp when the workflow started.
  </ResponseField>
  
  <ResponseField name="item.completedAt" type="string | null">
    ISO 8601 timestamp when completed (if applicable).
  </ResponseField>
  
  <ResponseField name="item.inputData" type="object">
    Original input data.
  </ResponseField>
  
  <ResponseField name="item.contextData" type="object">
    Current context data (may be modified during execution).
  </ResponseField>
  
  <ResponseField name="item.outputData" type="object">
    Final output data (populated on completion).
  </ResponseField>
  
  <ResponseField name="item.graphJson" type="object">
    The workflow graph structure.
  </ResponseField>
  
  <ResponseField name="item.builderLayout" type="object">
    Visual layout information.
  </ResponseField>
  
  <ResponseField name="item.steps" type="array">
    Array of all step instances executed in this workflow.
    
    <ResponseField name="item.steps[].id" type="UUID">
      Step instance ID.
    </ResponseField>
    
    <ResponseField name="item.steps[].stepCode" type="string">
      Step code identifier.
    </ResponseField>
    
    <ResponseField name="item.steps[].stepLabel" type="string">
      Human-readable step label.
    </ResponseField>
    
    <ResponseField name="item.steps[].stepType" type="string">
      Type: `start`, `human_task`, `subworkflow`, `end`, etc.
    </ResponseField>
    
    <ResponseField name="item.steps[].status" type="string">
      Step status: `pending`, `active`, `waiting`, `completed`, `rejected`, `reverted`, `failed`, `cancelled`.
    </ResponseField>
    
    <ResponseField name="item.steps[].visitCount" type="integer">
      Number of times this step has been visited (for loops).
    </ResponseField>
    
    <ResponseField name="item.steps[].enteredAt" type="string">
      When the step was entered.
    </ResponseField>
    
    <ResponseField name="item.steps[].completedAt" type="string | null">
      When the step completed.
    </ResponseField>
    
    <ResponseField name="item.steps[].waitingSince" type="string | null">
      When the step started waiting (for human tasks).
    </ResponseField>
    
    <ResponseField name="item.steps[].resultAction" type="string | null">
      The action that completed this step.
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="item.actions" type="array">
    All actions performed on this workflow.
    
    <ResponseField name="item.actions[].id" type="UUID">
      Action ID.
    </ResponseField>
    
    <ResponseField name="item.actions[].actionType" type="string">
      Type: `start`, `approve`, `reject`, `revert`, `custom`.
    </ResponseField>
    
    <ResponseField name="item.actions[].actionCode" type="string | null">
      Custom action code (for custom actions).
    </ResponseField>
    
    <ResponseField name="item.actions[].actorUserId" type="string">
      User who performed the action.
    </ResponseField>
    
    <ResponseField name="item.actions[].actorType" type="string">
      Actor type: `user` or `system`.
    </ResponseField>
    
    <ResponseField name="item.actions[].remarkText" type="string | null">
      Optional comment provided with the action.
    </ResponseField>
    
    <ResponseField name="item.actions[].payload" type="object">
      Action-specific payload data.
    </ResponseField>
    
    <ResponseField name="item.actions[].createdAt" type="string">
      When the action was performed.
    </ResponseField>
  </ResponseField>
</ResponseField>

## Example Request

<CodeGroup>

```bash cURL
curl -X GET https://api.example.com/api/v1/workflow-instances/123e4567-e89b-12d3-a456-426614174000 \
  -H "Authorization: Bearer YOUR_TOKEN"
```

```javascript JavaScript
const response = await fetch(
  'https://api.example.com/api/v1/workflow-instances/123e4567-e89b-12d3-a456-426614174000',
  {
    headers: {
      'Authorization': 'Bearer YOUR_TOKEN'
    }
  }
);

const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://api.example.com/api/v1/workflow-instances/123e4567-e89b-12d3-a456-426614174000',
    headers={'Authorization': 'Bearer YOUR_TOKEN'}
)

data = response.json()
```

</CodeGroup>

</ResponseField>

<ResponseField name="401" type="Unauthorized">
  Missing or invalid authentication token.
</ResponseField>

<ResponseField name="404" type="Not Found">
  Workflow instance not found.
</ResponseField>
