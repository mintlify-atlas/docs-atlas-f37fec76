---
title: Create Workflow Instance
api: POST /api/v1/workflow-instances
---

Starts a new workflow instance from a workflow definition.

## Authentication

Requires Bearer token authentication. The authenticated user will be recorded as the workflow initiator.

## Request Body

<ParamField body="workflowDefinitionId" type="UUID">
  The unique identifier of the workflow definition. Either this or `workflowKey` is required.
</ParamField>

<ParamField body="workflowKey" type="string">
  The key of the workflow definition. Either this or `workflowDefinitionId` is required.
</ParamField>

<ParamField body="businessKey" type="string">
  Optional business identifier for this workflow instance. Used for tracking and correlation purposes.
</ParamField>

<ParamField body="inputData" type="object" default="{}">
  Initial input data for the workflow. This data is available to all steps in the workflow.
</ParamField>

<ParamField body="contextData" type="object" default="{}">
  Initial context data for the workflow. Context data can be modified as the workflow progresses through steps.
</ParamField>

## Response

<ResponseField name="item" type="object">
  The created workflow instance with full details.
  
  <ResponseField name="item.id" type="UUID">
    Unique identifier for the workflow instance.
  </ResponseField>
  
  <ResponseField name="item.workflowDefinitionId" type="UUID">
    The workflow definition this instance belongs to.
  </ResponseField>
  
  <ResponseField name="item.workflowDefinitionKey" type="string">
    The key of the workflow definition.
  </ResponseField>
  
  <ResponseField name="item.workflowDefinitionName" type="string">
    Human-readable name of the workflow.
  </ResponseField>
  
  <ResponseField name="item.workflowVersionId" type="UUID">
    The specific version of the workflow definition being executed.
  </ResponseField>
  
  <ResponseField name="item.businessKey" type="string | null">
    The business key provided at creation time.
  </ResponseField>
  
  <ResponseField name="item.runNumber" type="integer">
    Sequential run number for this business key. Starts at 1 for each unique business key.
  </ResponseField>
  
  <ResponseField name="item.status" type="string">
    Current workflow status. One of: `running`, `waiting`, `paused`, `completed`, `rejected`, `failed`, `cancelled`.
  </ResponseField>
  
  <ResponseField name="item.currentStepCode" type="string | null">
    Code identifier of the current step.
  </ResponseField>
  
  <ResponseField name="item.currentStepLabel" type="string | null">
    Human-readable label of the current step.
  </ResponseField>
  
  <ResponseField name="item.startedBy" type="string">
    User ID of the workflow initiator.
  </ResponseField>
  
  <ResponseField name="item.startedAt" type="string">
    ISO 8601 timestamp when the workflow started.
  </ResponseField>
  
  <ResponseField name="item.completedAt" type="string | null">
    ISO 8601 timestamp when the workflow completed (if applicable).
  </ResponseField>
  
  <ResponseField name="item.inputData" type="object">
    The input data provided at creation.
  </ResponseField>
  
  <ResponseField name="item.contextData" type="object">
    Current context data (may be modified by workflow steps).
  </ResponseField>
  
  <ResponseField name="item.outputData" type="object">
    Output data (populated when workflow completes).
  </ResponseField>
  
  <ResponseField name="item.graphJson" type="object">
    The workflow graph structure.
  </ResponseField>
  
  <ResponseField name="item.builderLayout" type="object">
    Layout information for the workflow builder.
  </ResponseField>
  
  <ResponseField name="item.steps" type="array">
    Array of step instances that have been executed.
  </ResponseField>
  
  <ResponseField name="item.actions" type="array">
    Array of actions performed on the workflow.
  </ResponseField>
</ResponseField>

  -H "Authorization: Bearer YOUR_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "workflowKey": "expense-approval",
    "businessKey": "EXP-2024-001",
    "inputData": {
      "amount": 1500.00,

    }
  }'

## Status Codes

<ResponseField name="201" type="Created">
  Workflow instance created successfully.
</ResponseField>

<ResponseField name="401" type="Unauthorized">
  Missing or invalid authentication token.
</ResponseField>

<ResponseField name="404" type="Not Found">
  Workflow definition not found.
</ResponseField>

<ResponseField name="422" type="Unprocessable Entity">
  Invalid request. Either `workflowDefinitionId` or `workflowKey` must be provided.
</ResponseField>
